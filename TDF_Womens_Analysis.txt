AvgBackmarkerGap = 
CALCULATE(
    AVERAGE(TDFF_Riders_History[GapSeconds]),
    TDFF_Riders_History[RiderGroup] = "Backmarker"
)

AvgTopContender = 
CALCULATE(
    AVERAGE(TDFF_Riders_History[GapSeconds]),
    TDFF_Riders_History[RiderGroup] = "Top Contender"
)

AvgChaserGap = 
CALCULATE(
    AVERAGE(TDFF_Riders_History[GapSeconds]),
    TDFF_Riders_History[RiderGroup] = "Chaser"
)

AvgMidRankGap = 
CALCULATE(
    AVERAGE(TDFF_Riders_History[GapSeconds]),
    TDFF_Riders_History[RiderGroup] = "Mid Rank"
)

AvgBackmarkerGapFormatted = 
VAR TotalSeconds = [AvgBackmarkerGap]
VAR Hours = INT( TotalSeconds / 3600 )
VAR Minutes = INT( MOD( TotalSeconds, 3600 ) / 60 )
RETURN
FORMAT(Hours, "0") & "h " & FORMAT(Minutes, "00") & "m"

Mo-chen power bi guided project
Measures
1.Sales = SUM ([Fact_Sales[Sales_USD])
2.Quantity = SUM ([Fact_Sales[Quantity])
3.COGs = SUM(Fact_Sales[COGS_USD])
4.Gross Profit = Sales - COGS
5.PYTD_Sales =
    CALCULATE (
        [Sales],
        SAMEPERIODLASTYEAR ( d_Date[Date] ),
        d_Date[InPast] = TRUE
    )
6.PYTD_Quantity =
    CALCULATE (
        [Quantity],
        SAMEPERIODLASTYEAR ( d_Date[Date] ),
        d_Date[InPast] = TRUE
    )
7.PYTD_GrossProfit =
    CALCULATE (
        [Gross Profit],
        SAMEPERIODLASTYEAR ( d_Date[Date] ),
        d_Date[InPast] = TRUE
    )

8.YTD_Sales = TOTALYTD([Sales],f_PlantSales[Date_Time])
9.YTD_Quantity = TOTALYTD([Quantity],f_PlantSales[Date_Time])
10.YTD_GrossProfit = TOTALYTD([Gross Profit],f_PlantSales[Date_Time])

11.Slicer_PriorYearToDate =
VAR selected_value = SELECTEDVALUE(SLC_Values[Values])
VAR result = SWITCH(selected_value,
"Sales", [PYTD_Sales],
"Quantity",[PYTD_Quantity],
"Gross Profit",[PYTD_GrossProfit],
BLANK()
)
RETURN
result

12.Slicer_YearToDate =
VAR selected_value = SELECTEDVALUE(SLC_Values[Values])
VAR result = SWITCH(selected_value,
"Sales", [YTD_Sales],
"Quantity",[YTD_Quantity],
"Gross Profit",[YTD_GrossProfit],
BLANK()
)
RETURN
result
